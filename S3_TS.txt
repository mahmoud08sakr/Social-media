export const uploadFile = async ({
  store = StorageEnum.memory,
  Bucket = process.env.S3_BUCKET_NAME as string,
  path = "general",
  ACL = "private" as ObjectCannedACL,
  file,
}: {
  store?: StorageEnum;
  Bucket?: string;
  path?: string | undefined;
  ACL?: ObjectCannedACL;
  file: Express.Multer.File;
}): Promise<string> => {
  const command = new PutObjectCommand({
    Bucket,
    Key: `${
      process.env.APPLICATION_NAME
    }/${path}/${Date.now()}__${Math.random()}/${file.originalname}`,
    ACL,
    Body:
      store === StorageEnum.memory ? file.buffer : createReadStream(file.path),
    ContentType: file.mimetype,
  });

  await s3Config().send(command);
  if (!command.input?.Key) {
    throw new BadRequestException("Fail to upload", { file });
  }
  return command.input.Key;
};

=========================================for larger files more than 5GB=======================================================

4-npm i @aws-sdk/lib-storage   => for larger files more than 5GB


Question
Why we need   npm i @aws-sdk/lib-storage upload method   instead of  PutObjectCommand ?



Answer

- S3 expects to receive the entire file in one go. This works well for small files (e.g. <100 MB), but it becomes: Slow for large files

- Memory-intensive (uses your server’s RAM)

- Risky — if upload fails mid-way, the entire file must be resent

- Impossible for files larger than 5 GB, since PutObject has a limit


Question
How to implemt it?

Answer
import { Upload } from "@aws-sdk/lib-storage";

export const uploadLargeFile = async ({
  store = StorageEnum.memory,
  Bucket = process.env.S3_BUCKET_NAME as string,
  path = "general",
  ACL = "private" as ObjectCannedACL,
  file,
}: {
  store?: StorageEnum;
  Bucket?: string;
  path?: string | undefined;
  ACL?: ObjectCannedACL;
  file: Express.Multer.File;
}): Promise<CompleteMultipartUploadCommandOutput> => {
  const upload = new Upload({
    client: s3Config(),
    params: {
      Bucket,
      Key: `${
        process.env.APPLICATION_NAME
      }/${path}/${Date.now()}__${Math.random()}_250/${file.originalname}`,
      ACL,
      Body:
        store === StorageEnum.memory
          ? file.buffer
          : createReadStream(file.path),
      ContentType: file.mimetype,
    },
  });

  upload.on("httpUploadProgress", (progress) => {
    console.log(`File upload progress is ::: `, progress);
  });
  return await upload.done();
};

What it does:
-Splits the file into 5MB+ parts
-each part separately in parallel

	-Retries failed parts automatically

	-Resumes uploads (optionally) if interrupted

	-Much more efficient for anything +100MB

| `PutObjectCommand`   | `Upload` (Multipart Upload) |
| -------------------- | --------------------------- |
| Simple, direct       | Smart, scalable             |
| Good for small files | Best for large files        |
| No retries/resume    | Retries and parallel parts  |
| Max size = 5GB       | Max size = 5TB              |

  keysToDelete,
  Quiet = false,
}: {
  Bucket?: string;
  keysToDelete: string[];
  Quiet?: boolean;
}): Promise<DeleteObjectsCommandOutput> => {
  const mappedKeysToDelete: { Key: string }[] = keysToDelete.map((Key) => {
    return { Key };
  });

  const command = new DeleteObjectsCommand({
    Bucket,
    Delete: {
      Objects: mappedKeysToDelete,
      Quiet,
    },
  });
  const result = await s3Config().send(command);
  console.log(result);
  return result;
};

=======================================================================Pre signed URL
WHY?

| Category                    | Pros ✅                                                                  | Cons ❌                                                              |
| --------------------------- | ------------------------------------------------------------------------ | --------------------------------------------------------------------- |
|  Security                   | - Secure temporary access without exposing credentials                   | - Anyone with the URL can use it within the valid time                |
| Expiration Control          | - Can set custom expiration time (e.g., minutes or hours)                | - Cannot revoke URL early unless you delete the object                |
| Granular Permissions        | - Specify HTTP method (`GET`, `PUT`, etc.) and object key                | - Cannot apply fine-grained permissions beyond basic action + time    |
| No User IAM Needed**        | - Clients (e.g., browsers/mobile) don't need AWS credentials             | - Must securely deliver the URL to the client                         |
| Performance                 | - Direct access avoids server proxying (faster and less load on backend) | - No download/upload progress control unless handled on client        |
| Simplicity                  | - Simple to implement with AWS SDK                                       | - Incorrect configuration leads to broken or insecure links           |
| Monitoring                  | - No server overhead for file access                                     | - Harder to log or track usage without enabling CloudTrail or S3 logs |
| Use Case Flexibility        | - Works for uploads, downloads, limited sharing                          | - Not ideal for long-term or frequently revoked access control        |

export const createUploadPresignedUrl = async ({
  Bucket = process.env.S3_BUCKET_NAME as string,
  path = "general",
  originalname,
  mimetype,
  expiresIn = 3600,
}: {
  Bucket?: string;
  path?: string;
  originalname: string;
  mimetype: string;
  expiresIn?: number;
}) => {
  const Key =
    `${process.env.APPLICATION_NAME}/${path}/` +
    Date.now() +
    "-" +
    originalname;
  const command = new PutObjectCommand({
    Bucket,
    Key,
    ContentType: mimetype,
  });
  const url = await getSignedUrl(s3Config(), command, { expiresIn }); // expires in 30 min
  return { url, Key };
};

import { Request } from "express";
import multer, { FileFilterCallback } from "multer";
import { BadRequestException } from "../response/error.response";
import os from "node:os";

export enum StorageEnum {
  disk = "disk",
  memory = "memory",
}

export const fileValidation = {
  image: ["image/jpeg", "image/png", "image/gif"],
};

export const cloudFileUpload = ({
  store = StorageEnum.memory,
  validation = [],
  maxFileSizeMB = 2,
}: {
  store?: StorageEnum;
  validation: string[];
  maxFileSizeMB?: number;
}) => {
  const storage =
    store === StorageEnum.memory
      ? multer.memoryStorage()
      : multer.diskStorage({
          destination: os.tmpdir(), // temporary folder
          filename: (req, file, cb) =>
            cb(null, `${Date.now()}-${file.originalname}`),
        });


  function fileFilter(
    req: Request,
    file: Express.Multer.File,
    callback: FileFilterCallback
  ) {
    if (!validation.includes(file.mimetype)) {
      return callback(new BadRequestException("IN-valid file format"));
    }

    return callback(null, true);
  }


  return multer({
    fileFilter,
    limits: { fileSize: maxFileSizeMB * 1024 * 1024 },
    storage,
  });

  
};

import { MongoClient } from "mongodb";

const MONGO_URI = "mongodb://localhost:27017";
const DB_NAME = "Enhance_demo";

let client;

const connectDB = async () => {
    if (!client) {
        client = new MongoClient(MONGO_URI);
        await client.connect();
    }
    return client;
};

export const handler = async (event) => {
    const client = await connectDB();
    const db = client.db(DB_NAME);
    const requests = db.collection("bsa_requests");

    for (const record of event.Records) {
        try {
            let fullKey = decodeURIComponent(
                record.s3.object.key.replace(/\+/g, " ")
            );

            console.log("S3 key:", fullKey);

            const parts = fullKey.split("/");

            /*
              BSA_PROJECT / Request / customId / projectAttachments / file.jpg
            */

            const customId = parts[2];
            const attachmentType = parts[3];
            const fileName = parts[4];

            if (!customId || !attachmentType || !fileName) {
                console.log("Invalid key format:", fullKey);
                continue;
            }
            console.log({ fullKey });

            const result = await requests.updateOne(
                { customId },
                {
                    $set: {
                        [`${attachmentType}.$[elem].status`]: "uploaded",
                        updatedAt: new Date(),
                    },
                },
                {
                    arrayFilters: [
                        { "elem.key": fullKey }
                    ],
                }
            );

            console.log({
                customId,
                attachmentType,
                matched: result.matchedCount,
                modified: result.modifiedCount,
            });

        } catch (error) {
            console.error("Lambda error for record:", error);
        }
    }

    return { statusCode: 200 };
};
